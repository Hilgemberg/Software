-- Consulta dados do proprietario do sistema suas filiaisselect a.razaoSocial_Proprietario, a.nomeFantasia_Proprietario, a.CNPJ_Proprietario, b.descricao_TipoEmpresa, a.idFilial_proprietario,c.descricao_Status,d.Email, d.observacao_Emailfrom oficina.tblProprietario ajoin oficina.tblTipoEmpresa bon (b.id_TipoEmpresa =  a.idTipoEmpresa_Proprietario)join oficina.tblStatus con (c.id_Status = a.idStatus_Proprietario)join oficina.tblemail don (d.idDono_Email = a.id_Proprietario);
